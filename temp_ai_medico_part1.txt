def configurar_gemini():
    """
    Configura la API de Google Gemini con la clave de API.
    
    Returns:
        bool: True si se configurÃ³ correctamente, False si hay error
    """
    if not GEMINI_AVAILABLE:
        logger.error("Google Generative AI no estÃ¡ disponible")
        return False
    
    try:
        # Obtener API key desde settings o variables de entorno
        api_key = (
            getattr(settings, 'GOOGLE_GEMINI_API_KEY', None) or 
            getattr(settings, 'GOOGLE_API_KEY', None) or 
            getattr(settings, 'GEMINI_API_KEY', None) or
            os.environ.get('GOOGLE_GEMINI_API_KEY') or
            os.environ.get('GOOGLE_API_KEY') or
            os.environ.get('GEMINI_API_KEY')
        )
        
        if not api_key:
            logger.error("GOOGLE_API_KEY no estÃ¡ configurado en settings o variables de entorno")
            return False
        
        # Configurar Gemini
        genai.configure(api_key=api_key)
        logger.info("âœ“ Google Gemini configurado correctamente")
        return True
        
    except Exception as e:
        logger.error(f"Error al configurar Gemini: {e}", exc_info=True)
        return False
